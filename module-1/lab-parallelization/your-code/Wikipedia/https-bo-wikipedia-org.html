སྐྱེ་ལྡན་རྫས་སྦྱོར། སྐྱེ་དངོས་རིགས་མཐུན། སྦུབས་ཕྲན། སྣོད་བཅུད་རིག་པ། ནགས་ལས་བསྟན་བཅོས། རིགས་རྒྱུན་དཔྱད་ཞིབ་རིག་པ། རྒས་ཆོས་རིག་པ། ནད་རིགས་སྔོན་འགོག་རིག་པ། རྫས་འགྱུར་བསྡུ་ལེན། སྐྱེ་དངོས་ཕྲ་རབ། རྫས་རྡུལ་ཕྲ་མོ། དབང་རྩའི་མ་ལག་དཔྱད་རིག སྨན་སྦྱོར་རིག་པ། ལུས་ཁམས་རིག་པ། ཁྱབ་འགྱེད་དཔྱད་རིག དུག་ཤེད་དཔྱད་ཞིབ་རིག་པ། དུད་འགྲོའི་ཚན་རིག ཕྲ་སྲིན་དཔྱད་རིག དབང་པོའི་མ་ལག འབྱུང་ཁམས་རིག་པ། ས་འགུལ་རྒྱུན་ཤེས། གཞུང་ཐིག འབྱུང་འཕྲོད། ཁྱིམ་ཞག འཛམ་གླིང་རྒྱས་བཤད། ཧི་མ་ལ་ཡ། ཕུར་བུ། སའི་གོ་ལ། རླུང་། གཟའ་སྐར་རིག་པ། སྐར་མ་སྨིན་དྲུག གནམ་རིག་སྐར་དཔྱད། སྣང་སྲིད་རིག་པ། དགུ་ཚིགས་སྐྱ་མོ། རྒྱུ་སྐར། སྐར་མ། གཟའ་སྐར་རྒྱུ་འགྲོས། གནམ་གཤིས་རིག་པ། ཉི་མ། ཟླ་བ། སྐར་མ་མིག་དམར།། སྐར་མ་ཕུར་བུ། སྐར་མ་པ་སངས། སྐར་མ་སྤེན་པ། འཁོར་སྐར། སྤྱི་ཚོགས་ཚན་རིག སྤྱི་ཚོགས་རིག་པ། མི་རིགས་རིག་པ། མི་རྒྱུད་དཔྱད་རིག སེམས་ཁམས་རིག་པ། མིའི་རིགས་རིག་པ། དཔལ་འབྱོར་རིགས་གནས། ཆབ་སྲིད་རིག་པ། སྐད་རིགས་རིག་པ། དངོས་དཔྱད་རིག་པ། སྤྱི་ཚོགས་ས་ཁམས་རིག་པ། མི་འབོར་རིག་པ། སྡོམ་བརྩིས་རིག་པ། ཆོས་ལུགས་་རིག་པ། ཆོས་ལུགས་་གཤིབ་བསྡུ་རིག་པ། གྲངས་རིག གྲངས་རིག ཨང་གྲངས། མ་ཡའི་ཨང་གྲངས། ཨ་རབ་ཨང་གྲངས། བོད་ཀྱི་ཨང་གྲངས་བརྩི་ཐབས། རོ་མའི་ཨང་གྲངས། གྲངས་རིག་གཞུང་ལུག་གྲགས་ཅན། གྲངས་རིག་པ གྲངས་རིག་དང་བརྟེན་འབྱུང་། རྩིས་འཕྲུལ་གྱི་གྲངས་རིག་རྟེན་གཞི། སྒྱུ་རྩལ། ཟློས་གར། གླུ་དབྱངས། རི་མོ། སྐུ་བཟོ། བཟོ་བཀོད། རྩོམ་རིག འཁྲབ་སྟོན། གློག་བརྙན། བརྙན་འཕྲིན། པར་ལེན། ཕབ་སྐྲུན། སྒྱུ་རྩལ་དོགས་འདྲི། སྲོལ་རྒྱུན་དམངས་ཁྲོད་རིག་གནས། འབྱུང་རྩིས། ཐང་ག སྐུ་གསུང་ཐུགས་རྟེན། ངག་རྩོམ། མངོན་བརྗོད་དང་ཞེ་ས། སྲོལ་རྒྱུན་དམངས་གཞས། ཚིག་རྒྱག་གམ་གྱོད་བཤད། འཕྲོད་བསྟེན་དང་ནད་རིགས། སྨན་རྩིས་ཁང་། གློ་གཅོང་གི་ནད། ཁྲག་གཏར་བའི་གསོ་བཅོས། རྒས་སྔ་བའི་ནད། ནད་དུག་ཕོག་པའི་མཆིན་ནད། ནད་གསོ་ཐབས་ཤེས། ནད་ཡམས། རིག་གནས་ཀུན་བཏུས། == བོད་ཀྱི་ས་ཆ་སོ་སོ།== ལྷ་ས།  གཞིས་ཀ་རྩེ།   ལྷོ་ཁ།  ཆབ་མདོ།   མངའ་རིས།   ཉིང་ཁྲི།   ནག་ཆུ།   འདམ་གཞུང་།  རྒྱལ་རྩེ་   འགྲམ།   ལྷ་རྩེ།   རེབ་གོང་།   སྡེ་དགེ  དཀར་མཛེས།  རྔ་བ།  མགོ་ལོག  ཡུལ་ཤུལ།  སེམས་ཀྱི་ཉི་ཟླ།  བདེ་ཆེན།  ཟི་ལིང་།  མདོ་ལ།  གསེར་ཆེན།  ཁྲི་ཀ  གཅན་ཚ།   བ་ཡན།  ཡ་རྫི།  འབའ།  མང་རྫོང་།  གཏེར་ལེན་ཁ།  ཨ་ཆེན་གངས་རྒྱན།།  གཙོད།  རྨ་ཆུ།  ཀླུ་ཆུ།  བསང་ཆུ།  བ་ཙེ།  འབྲུག་ཆུ།  ཆུ་ཆེན་བཙན་ལྷ།  གཟི་རྩ་སྡེ་དགུ  རྒྱལ་རོང་།  མུ་ལེ།  ལྕགས་ཟམ་ཁ།   འཛམ་གླིང་ཆོས་ལུགས་སོ་སོ། ཡེ་ཤུའི་ཆོས་ལུགས - ཁ་ཆེའི་ཆོས་ལུགས།  • གནམ་བདག་ཆོས་ལུགས། • རྟོག་གེ་ཆོས་ལུགས། •བོན་ཆོས། • སངས་རྒྱས་ཆོས་ལུགས། •་ཧིན་རྡུ་ཆོས་ལུགས། • གཅེར་བུ་བའི་ཆོས་ལུགས། • བོད་བརྒྱུད་ནང་བསྟན། •རྒྱ་བརྒྱུད་ནང་བསྟན། •ལྷོ་རྒྱུད་ནང་བསྟན། •བོད་བརྒྱུད་ནང་བསྟན་གྱི་གྲུབ་མཐའ་སོ་སོ། • བོད་ཀྱི་གནའ་ཤུལ་གྲགས་ཅན། ཕོ་བྲང་ཡུམ་བུ་བླ་སྒང།   ཕོ་བྲང་པོ་ཏ་ལ།   ར་ས་གཙུག་ལག་ཁང།   བསམ་ཡས་གཙུག་ལག་ཁང་།   སྐུ་མཁར་མ་རུ།   རྒྱལ་རོང་བཙན་རྫོང་།   གུ་གེ་རྒྱལ་རབས་ཀྱི་གནའ་ཤུལ།   ཕོ་བྲང་རྒྱལ་མཁར་རྩེ།   ཏུན་ཧོང་བྲག་ཕུག།   ལྷོ་བྲག་སྲས་མཁར་དགུ་ཐོག།   ཁ་རུབ་ཀྱི་གནའ་ཤུལ།   བཙན་པོའི་བང་སོ།   འཕྱིང་བ་སྟག་རྩེའི་རྗེས་ཤུལ།   ར་མོ་ཆེ་གཙུག་ལག་ཁང།   རྒྱ་མ་སྒང་ཀྱི་གནའ་ཤུལ།   དཔལ་ལྡན་འབྲས་སྤུངས་དགོན།   སེ་རྭ་ཐིག་ཆེན་གླིང།   རྒྱལ་རྩེ་དཔལ་འཁོར་ཆོས་སྡེ།   བཀྲ་ཤིས་ལྷུན་པོ་དགོན།   ས་སྐྱ་དགོན།   མཚུར་ཕུ་དགོན།   འབྲོག་རི་བོ་དགའ་ལྡན་རྣམ་པར་རྒྱལ་བའི་གླིང།   ནོར་བུ་གླིང་ཁ།   རྫོང་རྒྱབ་ཀླུ་ཁང།   ཞང་ཞུང་རྒྱལ་རབས་ཀྱི་བང་སོ།   སྐུ་འབུམ་དགོན་པ།  བྱང་གི་དགོན་ཆེན་བཞི།  བླ་བྲང་བཀྲ་ཤིས་འཁྱིལ།  གཙོད་དགུ་ཐོག  སྲོང་བཙན་གླིང་།  གསེར་རྟ་དགོན་པ།  དུར་ལམ་གནའ་ཤུལ།  ཡུལ་ཤུལ་གནའ་ཤུལ།   ལིང་ཀྲིག་གནའ་ཤུལ།   ལུང་མདོ་དང་གཞོང་ཐང་གི་གནའ་ཤུལ།  བོད་ཀྱི་བཙན་པོ་རིམ་བྱོན། གཉའ་ཁྲི་བཙན་པོ།   མུ་ཁྲི་བཙན་པོ།   དེང་ཁྲི་བཙན་པོ།   སོ་ཁྲི་བཙན་པོ།   དར་ཁྲི་བཙན་པོ།   དྭགས་ཁྲི་བཙན་པོ།   སྲིབ་ཁྲི་བཙན་པོ།   གྲི་གུམ་བཙན་པོ།   སྤུ་ལྡེའུ་གུང་རྒྱལ།   ཨ་ཤོ་ལེགས་རྒྱལ།   དེ་ཤོ་ལེགས་རྒྱལ།   ཐི་ཤོ་ལེགས་རྒྱལ།   གུ་རུ་ལེགས་རྒྱལ།   འབྲོང་ཞི་ལེགས་རྒྱལ།   ཨི་ཤོ་ལེགས།   ཟ་ནམ་ཟིན་ལྡེ།   ལྡེ་འཕྲུལ་ནམ་གཞུང་བཙན།  སེ་སྣོལ་གནམ་ལྡེ།  ལྡེ་སྣོལ་ནམ།   ལྡེ་སྣོལ་པོ།   ལྡེ་རྒྱལ་པོ།   ལྡེ་སྤྲིན་བཙན། ཏོ་རི་ལུང་བཙན།  ཁྲི་བཙན་ནམ།   ཁྲི་སྒྲ་དཔུང་བཙན།   ཁྲི་ཐོག་རྗེ་བཙན།   ལྷ་ཐོ་ཐོ་རི་གཉན་བཙན།  ཁྲི་གཉན་གཟུང་བཙན།   འབྲོང་གཉན་ལྡེའུ།   སྟག་རི་གཉན་གཟིགས།   གནམ་རི་སྲོང་བཙན།  སྲོང་བཙན་སྒམ་པོ།   གུང་རི་གུང་བཙན།   མང་སྲོང་མང་བཙན།   འདུས་སྲོང་མང་པོ་རྗེ།   ཁྲི་ལྡེ་གཙུག་བརྟན།   ཁྲི་སྲོང་ལྡེའུ་བཙན།   མུ་ནེ་བཙན་པོ།   མུ་ཏིག་བཙན་པོ།   ཁྲི་ལྡེ་སྲོང་བཙན།   ཁྲི་གཙུག་ལྡེ་བཙན།   འུ་དུམ་བཙན་པོ།   གནམ་ལྡེ་འོད་སྲུང།   ཁྲི་ལྡེ་ཡུམ་བརྟན།   རྒྱལ་དབང་སྐུ་ཕྲེང་རིམ་བྱོན། རྒྱལ་བ་དགེ་འདུན་གྲུབ།   རྒྱལ་བ་དགེ་འདུན་རྒྱ་མཚོ།   རྒྱལ་བ་བསོད་ནམས་རྒྱ་མཚོ།   རྒྱལ་བ་ཡོན་ཏན་རྒྱ་མཚོ།   རྒྱལ་བ་བློ་བཟང་རྒྱ་མཚོ།   རྒྱལ་བ་ཚངས་དབྱངས་རྒྱ་མཚོ།   རྒྱལ་བ་སྐལ་བཟང་རྒྱ་མཚོ།   རྒྱལ་བ་འཇམ་དཔལ་རྒྱ་མཚོ།   རྒྱལ་བ་ལུང་རྟོགས་རྒྱ་མཚོ།   རྒྱལ་བ་ཚུལ་ཁྲིམས་རྒྱ་མཚོ།   རྒྱལ་བ་མཁས་གྲུབ་རྒྱ་མཚོ།   རྒྱལ་བ་འཕྲིན་ལས་རྒྱ་མཚོ།   རྒྱལ་བ་ཐུབ་བསྟན་རྒྱ་མཚོ།   རྒྱལ་བ་བསྟན་འཛིན་རྒྱ་མཚོ།   པཎ་ཆེན་སྐུ་ཕྲེང་རིམ་བྱོན། པཎ་ཆེན་སྐུ་ཕྲེང་དང་པོ་མཁས་གྲུབ་དགེ་ལེགས་དཔལ་བཟང་།   པཎ་ཆེན་སྐུ་ཕྲེང་གཉིས་པ་བསོད་ནམས་ཕྱོགས་ཀྱི་གླང་བོ།   པཎ་ཆེན་སྐུ་ཕྲེང་གསུམ་པ་དབེན་ས་བ་བློ་བཟང་དོན་འགྲུབ།   པཎ་ཆེན་སྐུ་ཕྲེང་བཞི་པ་བློ་བཟང་ཆོས་ཀྱི་རྒྱལ་མཚན།   པཎ་ཆེན་སྐུ་ཕྲེང་ལྔ་པ་བློ་བཟང་ཡེ་ཤེས།   པཎ་ཆེན་སྐུ་ཕྲེང་དྲུག་པ་དཔལ་ལྡན་ཡེ་ཤེས།   པཎ་ཆེན་སྐུ་ཕྲེང་བདུན་པ་བསྟན་པའི་དབང་ཕྱུག།   པཎ་ཆེན་སྐུ་ཕྲེང་བརྒྱད་པ་བསྟན་པའི་ཉི་མ།   པཎ་ཆེན་སྐུ་ཕྲེང་དགུ་པ་བློ་བཟང་ཐུབ་བསྟན་ཆོས་ཀྱི་ཉི་མ།
